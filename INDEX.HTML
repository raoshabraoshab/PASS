VPN PowerShell Script

$VPN_NAME = 'VPN' 
$SERVER_ADDRESS = 'uk205.vpnbook.com'
$USERNAME = 'vpnbook'
$PASSWORD = 'huba7re'

# Create the VPN connection
Add-VpnConnection -Name $VPN_NAME -ServerAddress $SERVER_ADDRESS -TunnelType PPTP -RememberCredential -PassThru

# Connect to the VPN
rasdial $VPN_NAME $USERNAME $PASSWORD
